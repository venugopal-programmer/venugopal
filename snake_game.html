What is Snake Game?

It is a popular game which is played by many people. But now it is not in trend. Anyhow, usually it is pre-installed in all keypad mobiles. We all played it in our childhood. In those days it is a great source of Entertainment and Timepass for us. We tried to break Hiscores.

In our game, there will be a Game Window, in which the game will executes. There will be a small snake and an apple whic is located on a random location on the Game Window. We have to control the snake by pressing Up arrow, Down arrow, Right arrow and left arroe buttons of our keyboard and try to overlap the head of the snake by the apple as we are feeding the snake. As we feed the snake The length of the snak will increases and the apple again locates at an random location of our game window. The Game Over function will returns true either if the head of the snake overlaps the body of the snake as it has bited itself or when the hea of the snake meets the border of the game window as it has hitted to the wall

    import pygame
    import random
    import os
    
    pygame.mixer.init()
    pygame.init()
    # colors
    white = (255, 255, 255)
    red = (255, 0, 0)
    black = (0, 0, 0)
    grey = (110, 110, 110)
    light_blue = (110, 189, 241)
    dark_blue = (0, 93, 155)
    # diplay area
    screen_width=900
    screen_height=500
    gameWindow = pygame.display.set_mode((screen_width, screen_height))
    
    #Background Image
    bgimg = pygame.image.load("snakebg.jpg")
    bgimg = pygame.transform.scale(bgimg, (screen_width, screen_height)).convert_alpha()
    # game title
    pygame.display.set_caption("Snake by VENUGOPAL")
    pygame.display.update()
    
    clock = pygame.time.Clock()
    font = pygame.font.SysFont(None, 55)
    
    
    def text_screen(text, color, x, y):
        screen_text = font.render(text, True, color)
        gameWindow.blit(screen_text, [x,y])
    
    def plot_snake(gameWindow, color, snk_list, snake_size):
        # print(snk_list)
        for x,y in snk_list:
            pygame.draw.rect(gameWindow, color, [x, y, snake_size, snake_size])
    
    def welcome():
        exit_game = False
        while not exit_game:
            gameWindow.fill(white)
            text_screen("Welcome To SNAKES by Venugopal", light_blue, 110, 190)
            text_screen("Press ENTER to start",grey , 230, 250)
            # text_screen("Hiscore: " + hiscore ,grey , 230, 250)
            for event in pygame.event.get():
                if event.type==pygame.QUIT:
                    exit_game=True
                if event.type==pygame.KEYDOWN:
                    if event.key==pygame.K_RETURN:
                        # pygame.mixer.music.load('F:\HP\Music\Jarico - Landscape [NCS BEST OF].mp3')
                        # pygame.mixer.music.play(-1)
                        gameloop()
            pygame.display.update()
            clock.tick(50)
    
    # game loop
    def gameloop():
        
            # Game specific varialbles
        exit_game = False
        game_over = False
        snake_x = 45
        snake_y = 55
        snake_size = 20
        fps= 20
        velocity_x = 0
        velocity_y = 0
        food_x = random.randint(40,screen_width-40)
        food_y = random.randint(40,screen_height-40)
        food_size = 20
        score = 0
    
        snk_list = []
        snk_length = 1
        # check if hiscore file exists
        
        with open("HighScoreOfSnakeGame.txt","r") as f:
            hiscore = f.read()
        
        while not exit_game:
            if game_over:
                with open("HighScoreOfSnakeGame.txt","w") as f:
                    f.write(str(hiscore))
                gameWindow.fill(white)
                text_screen(f'''Game Over! Press enter to continue''', red, 100, 200)
                text_screen("Score: " + str(score) + "  Hiscore : " + str(hiscore),red, 5,5)
                for event in pygame.event.get():
                    if event.type == pygame.QUIT:
                        exit_game = True
                    if event.type == pygame.KEYDOWN:
                        if event.key==pygame.K_RETURN:  
                            welcome()  
            else:    
                for event in pygame.event.get():
                    if event.type == pygame.QUIT:
                        exit_game = True
                    if event.type == pygame.KEYDOWN:
                        if event.key==pygame.K_RIGHT:
                            velocity_x = 7 
                            velocity_y = 0   
                        elif event.key==pygame.K_LEFT:
                            velocity_x = -7    
                            velocity_y = 0
                        elif event.key==pygame.K_UP:
                            velocity_y =  -7 
                            velocity_x = 0   
                
                        elif event.key==pygame.K_DOWN:
                            velocity_y = 7
                            velocity_x = 0 
                        # Below is cheat code    
                        elif event.key==pygame.K_h:
                            score+=5       
                
                snake_x = snake_x + velocity_x
                snake_y = snake_y + velocity_y
    
                if abs(snake_x - food_x)<20 and abs(snake_y - food_y)<20:
                    score +=10
                    # print("Score: ", score * 10)
                    food_x = random.randint(20, screen_width-10)
                    food_y = random.randint(20, screen_height-10)
                    snk_length += 8
                    if score>int(hiscore):
                        hiscore = score
    
                gameWindow.fill(white)
                gameWindow.blit(bgimg, (0, 0))
                text_screen("Score: " + str(score) + "  Hiscore : " + str(hiscore),dark_blue, 5,5)
                pygame.draw.rect(gameWindow, red, [food_x, food_y, food_size, food_size])
    
                head= []
                head.append(snake_x)
                head.append(snake_y)
                snk_list.append(head)
    
                if len(snk_list)>snk_length:
                    del snk_list[0]
    
                if head in snk_list[:-1]:
                    game_over = True    
    
    
                if snake_x<0 or snake_x>screen_width or snake_y<0 or snake_y>screen_height:
                    game_over=True
                    # print("Game Over")
                # pygame.draw.rect(gameWindow, black, [snake_x, snake_y, snake_size, snake_size])
                plot_snake(gameWindow, black, snk_list, snake_size)
            pygame.display.update()
            clock.tick(fps)
    
    
        pygame.quit()
        quit()
    welcome()
